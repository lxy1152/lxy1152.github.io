public int removeDuplicates(int[] nums) {
    int n = nums.length;
    if (n <= 1) {
        return n;
    }
    int cur = 0;
    for (int i = 0; i < n - 1; i++) {
        if (nums[i] != nums[i + 1]) {
            nums[cur++] = nums[i];
        }
    }
    nums[cur++] = nums[n - 1];
    return cur;
}
